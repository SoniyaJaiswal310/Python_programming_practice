num = int(input("enter the number of terms = "))
term1=0
term2=1
if num==1:
    print(f"Fibonacci = {term1}")
elif num==2:
        print(f"Fibonacci = {term1} {term2}")
else:
    print(f"Fibonacci = {term1} {term2}", end = " ")
    for i in range(2,num):
        fib = term1+term2
        term1=term2
        term2 = fib
        print(fib, end = "  ")
