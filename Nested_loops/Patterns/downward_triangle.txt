row = int(input("enter the number of rows = "))
symbol = input("enter the symbol = ")
for i in range(0,row):
    for j in range(0,i):
        print(" ", end = " ")
    for j in range(i,row):
         print(symbol, end = " ")
    for j in range(i,row-1):
        print(symbol, end = " ")
    print()
